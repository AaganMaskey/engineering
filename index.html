Write Test Case:

	
import { test } from 'qunit';
import moduleForAcceptance from 'fuseprospector-web/tests/helpers/module-for-acceptance';

moduleForAcceptance('Acceptance | overiden row checkbox should work');

	/**
		 * what is happening
		 * - checkbox is selected, but user is not given to "Mark lead as ready to connect"
		 * what should happen:
		 * - row changes color
		 * - actions menu appears (with a button "Mark as ready to connect" present)
		 *
		 * @NOTE this is a data bound test => only runs if there is at least one row with overiden errors
		 */
		test('click on the checkbox should trigger presence of move to ready to connect', assert => {
		  visit('/sourcing/clean?size=50');

			andThen(() => {
		    	if(prerequisites(find)) {
					// 1. make sure we are on the right page
					assert.equal(currentURL(), '/sourcing/clean?size=50', 'sourcing/clean page should be open');

					// 2. click on the checkbox
					let undoOverride = Ember.$(find('.ico.icon-undo')[0]);
					let rowCheckbox = undoOverride.parent().parent().parent().parent().children('.checkbox_fixes').children('.ui-checkbox');
					rowCheckbox.click();

					andThen(() => {
						// 3. check for the right correct behavior
						let row = undoOverride.parent().parent().parent().parent();
						// 3.a row changes the color
						assert.ok(row.hasClass( 'selected-row' ), 'row should have a "selected-row" css class');

						// 3.b "move to ready to connect" shows up
						let actionButton = find('.data-head-left-bar > .inline-list > .button-group');
						assert.ok(actionButton.html().search('Mark as Ready to Connect') >= 0, 'mark as ready to connect button should be present');

					});

				} else {
					assert.ok(true); // ignore this test if pre-requisite is not fulfilled
				}
		    });

		});
		/**
		 * Is there a row with undo button?
		 */
		function prerequisites(find) {
			let isContainsData = find('tbody tr').length > 0;
		    // look if it contains data or not
		    return isContainsData && !!(find('.ico.icon-undo')[0]);
		}
